product['stock'] = product['stock'].apply(lambda row : row.split(' ', 1)[0])
product['rating'] = product['rating'].replace('null', 'None')
product['reviews'] = product['reviews'].apply(lambda row : row.split(' ', 1)[0][1:]).replace('', '0').astype(int)
